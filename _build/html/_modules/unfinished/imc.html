[docs]def finite_run(I: Predicate, T: Predicate, P: Predicate, k: int):
    F = Predicate(~P[0])
    if is_sat(I[0] & F[0]):
        return Status.UNSAFE1
    R = I[0]
    while True:
        A = R & T[0]
        B = T[1:k - 1] & Or(F[l] for l in range(k + 1))
        if is_sat(A & B):
            if is_valid(EqualsOrIff(R, I[0])):
                return Status.UNSAFE2
            else:
                return Status.UNKNOWN
        else:
            P = Predicate(binary_interpolant(A, B))
            R_prime = P[0]
            if is_valid(R_prime.Implies(R)):
                return Status.SAFE
            R |= R_prime
